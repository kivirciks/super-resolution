def download_archive(file, target_dir, extract=True):
    source_url = f'http://data.vision.ee.ethz.ch/cvl/DIV2K/{file}'
    target_dir = os.path.abspath(target_dir)

    os.remove(os.path.join(target_dir, file))

```python
# –û—Ü–µ–Ω–∫–∞ –∑–Ω–∞—á–µ–Ω–∏—è PSNR: PSNR = 20 * log10(MAXp) - 10 * log10(MSE)
def PSNR(y_true, y_pred, MAXp=1):
    # y_true - —Ä–µ–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
    # y_pred - –ø—Ä–µ–¥—Å–∫–∞–∑—ã–≤–∞–µ–º–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ
    # MAXp - –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ –¥–∏–∞–ø–∞–∑–æ–Ω–∞ –ø–∏–∫—Å–µ–ª–µ–π (–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é=1).
    return -10.0 * K.log(K.mean(K.square(y_pred - y_true))) / K.log(10.0)
```
